func minSubArrayLen(target int, nums []int) int {
	length := len(nums)
	ans := length + 1
	left := 0
	prefixSum := 0
	for right := 0; right < length; right++ {
		//prefixSum = nums[0]+...+nums[i]
		prefixSum += nums[right]
		for prefixSum >= target {
			ans = min(ans, right-left+1)
            prefixSum -= nums[left]
			left++
		}
	}
	if ans == length+1 {
		return 0
	} else {
		return ans
	}
}
