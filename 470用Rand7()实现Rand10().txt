func rand10() int {
	/*
    randx():等概率生成1~x
    1,2,...,x
    randx()-1:等概率生成0~x-1
    0,1,...,x-1
    (randx()-1)*y:等概率生成0;y;...;(x-1)y
    randy():等概率生成
    1,2,3,...,y
    (randx()-1)*y+randy():等概率生成1~xy
    1,2,...,y;y+1,y+2,...,2y;...;(x-1)y+1,(x-1)y+2,...,xy
    */
	for {
		x := rand7()
        y := rand7()
        //等概率生成1~49
		n := (x-1)*7 + y
        //取等概率生成的1~40
		if n <= 40 {
            //1,11,21,31->1;2,12,22,32->2 依此类推
			return (n - 1) % 10 + 1
		}
	}
}
