func subsets(nums []int) [][]int {
    var ans [][]int
    var path []int
    var backtracking func(startIndex int)
    backtracking = func(startIndex int) {
        temp := append([]int{}, path...)
        ans = append(ans, temp)
        for i := startIndex; i < len(nums); i++ {
            path = append(path, nums[i])
            backtracking(i + 1)
            path = path[:len(path)-1]
        }
    }
    backtracking(0)
    return ans
}
