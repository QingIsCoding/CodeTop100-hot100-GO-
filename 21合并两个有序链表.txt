/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func mergeTwoLists(list1 *ListNode, list2 *ListNode) *ListNode {
    cur := &ListNode{}
    dummyHead := cur
    for list1 != nil && list2 != nil {
        if list1.Val < list2.Val {
            cur.Next = list1
            cur = cur.Next
            list1 = list1.Next
        } else {
            cur.Next = list2
            cur = cur.Next
            list2 = list2.Next
        }
    }
    for list1 != nil {
        cur.Next = list1
        cur = cur.Next
        list1 = list1.Next
    }
    for list2 != nil {
        cur.Next = list2
        cur = cur.Next
        list2 = list2.Next
    }
    return dummyHead.Next
}
